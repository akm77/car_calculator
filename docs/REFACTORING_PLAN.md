def _utilization_fee_v2(
    age_category: str,
    engine_cc: int,
    engine_power_hp: int,
    rates_conf: dict[str, Any]
) -> tuple[Decimal, float]:
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç: (—Å—É–º–º–∞_—Ä—É–±–ª–µ–π, –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç)

    2. –ù–∞–π—Ç–∏ volume_band –ø–æ engine_cc
    3. –í–Ω—É—Ç—Ä–∏ band –Ω–∞–π—Ç–∏ power_bracket –ø–æ kw
    4. –í—ã–±—Ä–∞—Ç—å coefficient_lt3 –∏–ª–∏ coefficient_gt3 –ø–æ age_category
    5. –†–∞—Å—Å—á–∏—Ç–∞—Ç—å: base_rate √ó coefficient
    """
    util = rates_conf.get("utilization_m1_personal", {})
    base_rate = to_decimal(util.get("base_rate_rub", 20000))
    
    # 1. –ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è
    engine_power_kw = engine_power_hp * 0.7355
    
    # 2. –ü–æ–∏—Å–∫ volume_band
    volume_bands = util.get("volume_bands", [])
    volume_band = None
    for band in volume_bands:
        vol_range = band.get("volume_range", [])
        if len(vol_range) == 2:
            if vol_range[0] <= engine_cc <= vol_range[1]:
                volume_band = band
                break
        else:  # –ø–æ—Å–ª–µ–¥–Ω–∏–π –¥–∏–∞–ø–∞–∑–æ–Ω (> max)
            if engine_cc > vol_range[0]:
                volume_band = band
                break
    
    if not volume_band:
        logger.warning(f"Volume band not found for {engine_cc} cc")
        return Decimal("0"), 0.0
    
    # 3. –ü–æ–∏—Å–∫ power_bracket
    power_brackets = volume_band.get("power_brackets", [])
    coefficient = 0.0
    for bracket in power_brackets:
        max_kw = bracket.get("power_kw_max")
        if max_kw is None or engine_power_kw <= max_kw:
            coef_key = "coefficient_lt3" if age_category == "lt3" else "coefficient_gt3"
            coefficient = to_decimal(bracket.get(coef_key, 0))
            break
    
    # 4. –†–∞—Å—á—ë—Ç
    fee = quantize4(base_rate * coefficient)
    return fee, float(coefficient)
